eps = 0.001;
lam = 1.7;
t1 = 120;
t2 = 340;
nMax = 1000;
h = 1.0;

x1 = 25;
i1 = IntegerPart[x1/h];

y1 = 8;
y2 = 12;
y3 = 16;
j1 = IntegerPart[y1/h];
j2 = IntegerPart[y2/h];
j3 = IntegerPart[y3/h];

deltaMax = eps + 1;
temp = Table[0, {i, i1*j3}];
temp = Partition[temp, i1];

For[j = 1, j <= j2, j++, temp[[j, 1]] = t1];
For[j = j1, j <= j3, j++, temp[[j, i1]] = t2];
kolit = 0;

While[(deltaMax > eps) && (kolit < nMax), deltaMax = 0; kolit++;
  For[i = 2, i <= i1, i++,
   For[j = 1, j < j1, j++,
    If[(i == i1) && (j1 > j > 1),
     tmp1 = 
      0.5*(0.5*(temp[[j + 1, i]] + temp[[j - 1, i]]) + 
         temp[[j, i - 1]])];
    If[(j == 1) && (i1 > i > 1),
     tmp1 = 
      0.5*(0.5*(temp[[j, i - 1]] + temp[[j, i + 1]]) + 
         temp[[j + 1, i]])];
    If[(j == 1) && (i == i1),
     tmp1 = 0.5*(temp[[j, i - 1]] + temp[[j + 1, i]])];
    If[(j1 > j > 1) && (i1 > i > 1),
     tmp1 = 
      0.25*(temp[[j, i - 1]] + temp[[j, i + 1]] + temp[[j - 1, i]] + 
         temp[[j + 1, i]])];
    delta = lam*(tmp1 - temp[[j, i]]);
    temp[[j, i]] = temp[[j, i]] + delta;
    deltaAbs = Abs[delta];
    If[deltaAbs > deltaMax, deltaMax = deltaAbs];
    ]
   ];
  For[i = 2, i < i1, i++,
   For[j = j1, j <= j2, j++,
    If[(j2 + 1 > j > j1 - 1) && (i1 > i > 1),
     tmp1 = 
      0.25*(temp[[j, i - 1]] + temp[[j, i + 1]] + temp[[j - 1, i]] + 
         temp[[j + 1, i]])];
    delta = lam*(tmp1 - temp[[j, i]]);
    temp[[j, i]] = temp[[j, i]] + delta;
    deltaAbs = Abs[delta];
    If[deltaAbs > deltaMax, deltaMax = deltaAbs];
    
    ]
   ];
  For[i = 1, i < i1, i++,
   For[j = j2 + 1, j <= j3, j++,
    If[(i == 1) && (j3 > j > j2),
     tmp1 = 
      0.5*(0.5*(temp[[j + 1, i]] + temp[[j - 1, i]]) + 
         temp[[j, i + 1]])];
    If[(j == j3) && (i1 > i > 1),
     tmp1 = 
      0.5*(0.5*(temp[[j, i - 1]] + temp[[j, i + 1]]) + 
         temp[[j - 1, i]])];
    If[(j3 > j > j2) && (i1 > i > 1),
     tmp1 = 
      0.25*(temp[[j, i - 1]] + temp[[j, i + 1]] + temp[[j - 1, i]] + 
         temp[[j + 1, i]])];
    If[(j == j3) && (i == 1),
     tmp1 = 0.5*(temp[[j, i + 1]] + temp[[j - 1, i]])];
    
    delta = lam*(tmp1 - temp[[j, i]]);
    temp[[j, i]] = temp[[j, i]] + delta;
    deltaAbs = Abs[delta];
    If[deltaAbs > deltaMax, deltaMax = deltaAbs];
    ]
   ];
  
  Print["Номер итерации = ", kolit, "  deltaMax=", deltaMax];
  ];
ListPlot3D[temp, ViewPoint -> {1.2, 1.2, 1.2}]

f1 = 0.5*(temp[[1, 1]] - temp[[1, 2]] + temp[[j2, 1]] - temp[[j2, 2]]);
For[j = 2, j < j2, j++, f1 = f1 + (temp[[j, 1]] - temp[[j, 2]])];


f2 = 0.5*(temp[[j1, i1]] - temp[[j1, i1 - 1]] + temp[[j3, i1]] - 
     temp[[j3, i1 - 1]]);
For[j = j1 + 1, j < j3, j++, 
  f2 = f2 + (temp[[j, i1]] - temp[[j, i1 - 1]])];

kons = Abs[f1 + f2]/(Abs[f1] + Abs[f2]);
Print["Итераций = ", kolit, "      f1=", f1, "  f2=", f2, "  kons= ", 
  kons];
